## 📌 개요

FMEA 온프레미스 시스템을 Docker 컨테이너로 패키징하여 기업 내부 서버에 배포하는 가이드입니다.

### 주요 특징
- ✅ 완전한 격리 환경 제공
- ✅ 간편한 설치 및 업데이트
- ✅ 확장 가능한 아키텍처
- ✅ 백업 및 복원 용이
- ✅ 모니터링 및 로깅 통합

## 📦 사전 준비사항

### 1. Docker 설치

#### Ubuntu/Debian
```bash
# Docker 설치
curl -fsSL https://get.docker.com -o get-docker.sh
sudo sh get-docker.sh

# Docker Compose 설치
sudo curl -L "https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

# 사용자 권한 설정
sudo usermod -aG docker $USER
```

#### CentOS/RHEL
```bash
# Docker 설치
sudo yum install -y yum-utils
sudo yum-config-manager --add-repo https://download.docker.com/linux/centos/docker-ce.repo
sudo yum install docker-ce docker-ce-cli containerd.io docker-compose-plugin

# Docker 시작
sudo systemctl start docker
sudo systemctl enable docker
```

### 2. 프로젝트 준비

```bash
# 프로젝트 클론
git clone <repository-url> fmea-onpremise
```

## 🚀 단계별 배포 가이드

### Step 1: Dockerfile 생성

```dockerfile
# Dockerfile
FROM node:18-alpine AS deps
RUN apk add --no-cache libc6-compat
WORKDIR /app
COPY package.json package-lock.json ./
RUN npm ci --only=production

FROM node:18-alpine AS builder
WORKDIR /app
COPY --from=deps /app/node_modules ./node_modules
COPY . .
ENV NEXT_TELEMETRY_DISABLED 1
RUN npm run build

FROM node:18-alpine AS runner
WORKDIR /app
ENV NODE_ENV production
ENV NEXT_TELEMETRY_DISABLED 1
RUN addgroup --system --gid 1001 nodejs
RUN adduser --system --uid 1001 nextjs
COPY --from=builder /app/public ./public
COPY --from=builder --chown=nextjs:nodejs /app/.next/standalone ./
COPY --from=builder --chown=nextjs:nodejs /app/.next/static ./.next/static
USER nextjs
EXPOSE 3000
ENV PORT 3000
CMD ["node", "server.js"]
```

### Step 2: Docker Compose 설정

```yaml
# docker-compose.yml
version: '3.8'

services:
  # PostgreSQL 데이터베이스
  postgres:
    image: postgres:15-alpine
    container_name: fmea-db
    restart: unless-stopped
    environment:
      POSTGRES_USER: ${DB_USER:-fmea_admin}
      POSTGRES_PASSWORD: ${DB_PASSWORD:-secure_password}
      POSTGRES_DB: ${DB_NAME:-fmea_db}
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./scripts/init-db.sql:/docker-entrypoint-initdb.d/init.sql
    ports:
      - "5432:5432"
    networks:
      - fmea-network
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${DB_USER:-fmea_admin}"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis (세션 저장소)
  redis:
    image: redis:7-alpine
    container_name: fmea-redis
    restart: unless-stopped
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    networks:
      - fmea-network
    command: redis-server --appendonly yes
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  # FMEA 애플리케이션
  app:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: fmea-app
    restart: unless-stopped
    environment:
      DATABASE_URL: postgresql://${DB_USER}:${DB_PASSWORD}@postgres:5432/${DB_NAME}
      REDIS_URL: redis://redis:6379
      NEXTAUTH_URL: ${APP_URL:-http://localhost}
      NEXTAUTH_SECRET: ${NEXTAUTH_SECRET}
      SMTP_HOST: ${SMTP_HOST}
      SMTP_PORT: ${SMTP_PORT}
      SMTP_USER: ${SMTP_USER}
      SMTP_PASS: ${SMTP_PASS}
    volumes:
      - ./uploads:/app/uploads
      - ./logs:/app/logs
    ports:
      - "3000:3000"
    networks:
      - fmea-network
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy

  # Nginx 리버스 프록시
  nginx:
    image: nginx:alpine
    container_name: fmea-nginx
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/sites:/etc/nginx/sites-enabled:ro
      - ./ssl:/etc/nginx/ssl:ro
      - nginx_logs:/var/log/nginx
    networks:
      - fmea-network
    depends_on:
      - app

volumes:
  postgres_data:
    driver: local
  redis_data:
    driver: local
  nginx_logs:
    driver: local

networks:
  fmea-network:
    driver: bridge
```

### Step 4: 배포 스크립트

```bash
#!/bin/bash
# deploy.sh - 배포 스크립트

echo "🚀 FMEA 시스템 배포 시작..."

# 1. 환경 변수 로드
export $(cat .env.production | xargs)

# 2. 이전 컨테이너 정지
echo "📦 기존 컨테이너 정지..."
docker-compose down

# 3. 이미지 빌드
echo "🔨 Docker 이미지 빌드..."
docker-compose build --no-cache

# 4. 데이터베이스 마이그레이션
echo "📊 데이터베이스 마이그레이션..."
docker-compose run --rm app npx prisma migrate deploy

# 5. 컨테이너 시작
echo "🎯 컨테이너 시작..."
docker-compose up -d

# 6. 헬스체크
echo "🏥 헬스체크 진행..."
sleep 10
curl -f http://localhost:3000/api/health || exit 1

echo "✅ 배포 완료!"
```

### Step 5: 백업 스크립트

```bash
#!/bin/bash
# backup.sh - 백업 스크립트

BACKUP_DIR="./backups"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)

# 백업 디렉토리 생성
mkdir -p $BACKUP_DIR

# 데이터베이스 백업
echo "📦 데이터베이스 백업 중..."
docker exec fmea-db pg_dump -U fmea_admin fmea_db > $BACKUP_DIR/db_backup_$TIMESTAMP.sql

# 업로드 파일 백업
echo "📁 파일 백업 중..."
tar -czf $BACKUP_DIR/files_backup_$TIMESTAMP.tar.gz ./uploads

# Redis 백업
echo "💾 Redis 백업 중..."
docker exec fmea-redis redis-cli SAVE
docker cp fmea-redis:/data/dump.rdb $BACKUP_DIR/redis_backup_$TIMESTAMP.rdb

echo "✅ 백업 완료: $BACKUP_DIR"
```

### 1. SSL/TLS 설정
```nginx
server {
    listen 443 ssl http2;
    server_name fmea.yourcompany.com;

    ssl_certificate /etc/nginx/ssl/cert.pem;
    ssl_certificate_key /etc/nginx/ssl/key.pem;
```
